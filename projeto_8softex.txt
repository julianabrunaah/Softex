function insertion_sort(n,arr) {
    let atual;
    let p;
    for(let i=1; arr.length; i++){
        atual = arr[i];
        p=i-1;
        while (p>=0 && arr[p]>atual) {
            arr[p+1] = arr[p]
            p--;
        }
        arr[p+1] = atual;
    }
    return arr;
}
